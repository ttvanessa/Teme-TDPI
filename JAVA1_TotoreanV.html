<!DOCTYPE html>
<html>
<head>
    <title>Calculator</title>
</head>
<body>
    <input type="number" id="num1" placeholder="Introduceți primul număr">
    <input type="number" id="num2" placeholder="Introduceți al doilea număr">
    <button onclick="multiply()">Înmulțire</button>
    <button onclick="divide()">Împărțire</button>
    <p id="result"></p>

    <script>
        function multiply() {
            const num1 = parseFloat(document.getElementById('num1').value);
            const num2 = parseFloat(document.getElementById('num2').value);
            const result = num1 * num2;
            document.getElementById('result').innerText = `Rezultatul înmulțirii: ${result}`;
        }

        function divide() {
            const num1 = parseFloat(document.getElementById('num1').value);
            const num2 = parseFloat(document.getElementById('num2').value);
            if (num2 !== 0) {
                const result = num1 / num2;
                document.getElementById('result').innerText = `Rezultatul împărțirii: ${result}`;
            } else {
                document.getElementById('result').innerText = "Nu se poate împărți la zero!";
            }
        }
    </script>
</body>
</html>
